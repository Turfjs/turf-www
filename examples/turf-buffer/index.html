turf-buffer

Buffers a point, linestring, or polygon Feature/FeatureCollection to a given radius. Units supported are miles, kilometers, and degrees.

function calculateBuffer() {
    var fc = turf.featurecollection(fcLayer.getLayers().map(function(f){
            return f.toGeoJSON()
        }));
    var buffer = turf.buffer(fc, 100, 'miles');
    buffer.properties = {
        "fill": "#6BC65F",
        "stroke": "#25561F",
        "stroke-width": 2
    };
    bufferLayer.setGeoJSON(buffer);
