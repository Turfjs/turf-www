turf-grid

Takes a bounding box and a cell depth and outputs a feature collection of points in a grid.

<div id='map'></div>

<script>
L.mapbox.accessToken = 'pk.eyJ1IjoibW9yZ2FuaGVybG9ja2VyIiwiYSI6Ii1zLU4xOWMifQ.FubD68OEerk74AYCLduMZQ';
var bbox = [-97.52, 35.460, -97.5, 35.48];
var bbox_poly = turf.bboxPolygon(bbox);
bbox_poly.properties = {
    "fill": "#6BC65F",
    "stroke": "#6BC65F",
    "stroke-width": 5
};

var grid = turf.grid(bbox, 5);


grid.features.forEach(function (feature) {
    feature.properties["marker-color"] = "#6BC65F";
    feature.properties["marker-size"] = "small";
});

var map = L.mapbox.map('map', 'morganherlocker.kgidd73k')
    .setView([35.471453, -97.508014], 14);

L.mapbox.featureLayer().setGeoJSON(bbox_poly).addTo(map);
L.mapbox.featureLayer().setGeoJSON(grid).addTo(map);
</script>
